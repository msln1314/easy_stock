# 卖出预警设计文档

## 1. 功能概述

卖出预警模块用于监控持仓股票的技术指标变化，当满足预设的预警条件时，将股票加入卖出预警股票池，提醒用户关注潜在的风险信号。

## 4. 指标计算函数实现

### 4.1 MA - 移动平均线

```python
def calc_ma(klines: dict, period: int = 5) -> dict:
    """
    计算移动平均线

    Args:
        klines: K线数据
        period: 计算周期

    Returns:
        {'values': [值列表], 'latest': 最新值}
    """
    closes = klines['close']
    values = []

    for i in range(len(closes)):
        if i < period - 1:
            values.append(None)
        else:
            values.append(sum(closes[i-period+1:i+1]) / period)

    return {
        'values': values,
        'latest': values[-1]
    }
```

### 4.2 EMA / EXPMA - 指数移动平均

```python
def calc_ema(klines: dict, period: int = 12) -> dict:
    """
    计算EMA指数移动平均

    Args:
        klines: K线数据
        period: 计算周期

    Returns:
        {'values': [值列表], 'latest': 最新值}
    """
    closes = klines['close']
    k = 2 / (period + 1)

    values = [sum(closes[:period]) / period]  # SMA作为初始值

    for i in range(period, len(closes)):
        values.append(closes[i] * k + values[-1] * (1 - k))

    # 前面补None对齐
    values = [None] * (period - 1) + values

    return {
        'values': values,
        'latest': values[-1]
    }

# EXPMA与EMA相同
calc_expma = calc_ema
```

### 4.3 BOLL - 布林带

```python
import statistics

def calc_boll(klines: dict, period: int = 20, std_dev: float = 2.0) -> dict:
    """
    计算布林带

    Args:
        klines: K线数据
        period: 计算周期
        std_dev: 标准差倍数

    Returns:
        {'upper': [...], 'middle': [...], 'lower': [...], 'latest': {...}}
    """
    closes = klines['close']

    upper = []
    lower = []
    middle = []

    for i in range(len(closes)):
        if i < period - 1:
            upper.append(None)
            lower.append(None)
            middle.append(None)
        else:
            std = statistics.stdev(closes[i-period+1:i+1])
            ma = sum(closes[i-period+1:i+1]) / period
            middle.append(ma)
            upper.append(ma + std_dev * std)
            lower.append(ma - std_dev * std)

    return {
        'upper': upper,
        'middle': middle,
        'lower': lower,
        'latest': {
            'upper': upper[-1],
            'middle': middle[-1],
            'lower': lower[-1]
        }
    }
```

### 4.4 MACD - MACD指标

```python
def calc_macd(klines: dict, fast: int = 12, slow: int = 26, signal: int = 9) -> dict:
    """
    计算MACD指标

    Args:
        klines: K线数据
        fast: 快线周期
        slow: 慢线周期
        signal: 信号线周期

    Returns:
        {'dif': [...], 'dea': [...], 'macd_bar': [...], 'latest': {...}}
    """
    closes = klines['close']

    # 计算快慢EMA
    ema_fast = calc_ema({'close': closes}, fast)['values']
    ema_slow = calc_ema({'close': closes}, slow)['values']

    # DIF线
    dif = [f - s if f is not None and s is not None else None
           for f, s in zip(ema_fast, ema_slow)]

    # DEA线（DIF的EMA）
    # 过滤None值计算DEA
    dif_valid = [d for d in dif if d is not None]
    k = 2 / (signal + 1)
    dea_valid = [sum(dif_valid[:signal]) / signal]
    for i in range(signal, len(dif_valid)):
        dea_valid.append(dif_valid[i] * k + dea_valid[-1] * (1 - k))

    # 对齐DEA
    none_count = len(dif) - len(dea_valid)
    dea = [None] * none_count + dea_valid

    # MACD柱
    macd_bar = [(d - de) * 2 if d is not None and de is not None else None
                for d, de in zip(dif, dea)]

    return {
        'dif': dif,
        'dea': dea,
        'macd_bar': macd_bar,
        'latest': {
            'dif': dif[-1],
            'dea': dea[-1],
            'macd_bar': macd_bar[-1]
        }
    }
```

### 4.5 KDJ - KDJ指标

```python
def calc_kdj(klines: dict, n: int = 9, m1: int = 3, m2: int = 3) -> dict:
    """
    计算KDJ指标

    Args:
        klines: K线数据
        n: RSV周期
        m1: K值平滑周期
        m2: D值平滑周期

    Returns:
        {'k': [...], 'd': [...], 'j': [...], 'latest': {...}}
    """
    highs = klines['high']
    lows = klines['low']
    closes = klines['close']

    k_values = []
    d_values = []
    j_values = []

    # 前面补None
    for _ in range(n - 1):
        k_values.append(None)
        d_values.append(None)
        j_values.append(None)

    for i in range(n - 1, len(closes)):
        high_n = max(highs[i-n+1:i+1])
        low_n = min(lows[i-n+1:i+1])

        # RSV值
        if high_n != low_n:
            rsv = (closes[i] - low_n) / (high_n - low_n) * 100
        else:
            rsv = 50

        # K值
        if i == n - 1:
            k = rsv
        else:
            k = k_values[-1] * (m1-1)/m1 + rsv * 1/m1

        # D值
        if i == n - 1:
            d = k
        else:
            d = d_values[-1] * (m2-1)/m2 + k * 1/m2

        # J值
        j = 3 * k - 2 * d

        k_values.append(k)
        d_values.append(d)
        j_values.append(j)

    return {
        'k': k_values,
        'd': d_values,
        'j': j_values,
        'latest': {
            'k': k_values[-1],
            'd': d_values[-1],
            'j': j_values[-1]
        }
    }
```

### 4.6 RSI - RSI指标

```python
def calc_rsi(klines: dict, period: int = 6) -> dict:
    """
    计算RSI指标

    Args:
        klines: K线数据
        period: 计算周期

    Returns:
        {'values': [...], 'latest': 最新值}
    """
    closes = klines['close']
    values = [None] * period  # 前面补None

    for i in range(period, len(closes)):
        changes = [closes[j] - closes[j-1] for j in range(i-period+1, i+1)]
        gains = sum([c for c in changes if c > 0])
        losses = sum([-c for c in changes if c < 0])

        if losses == 0:
            rsi = 100
        else:
            rs = gains / losses
            rsi = 100 - 100 / (1 + rs)

        values.append(rsi)

    return {
        'values': values,
        'latest': values[-1]
    }
```

### 4.8 VOL_RATIO - 量比

```python
def calc_vol_ratio(klines: dict, period: int = 5) -> dict:
    """
    计算量比

    Args:
        klines: K线数据
        period: 均量周期

    Returns:
        {'value': 量比值}
    """
    volumes = klines['volume']

    if len(volumes) < period:
        return {'value': None}

    avg_volume = sum(volumes[-period:]) / period
    current_volume = volumes[-1]
    ratio = current_volume / avg_volume

    return {
        'value': ratio
    }
```

## 7. 预警评估服务

```python
import json
from typing import Dict, Any

class WarningEvaluator:
    """预警条件评估服务"""

    def __init__(self):
        self.calculator = IndicatorCalculator()

    def evaluate(self, klines: dict, condition: dict) -> tuple[bool, dict]:
        """
        评估预警条件

        Args:
            klines: K线数据
            condition: 预警条件配置

        Returns:
            (是否触发, 触发时的指标值)
        """
        rule = json.loads(condition['condition_rule'])

        if rule['rule_type'] == 'cross':
            return self._evaluate_cross(klines, condition, rule)

        elif rule['rule_type'] == 'break':
            return self._evaluate_break(klines, condition, rule)

        elif rule['rule_type'] == 'threshold':
            return self._evaluate_threshold(klines, condition, rule)

        return False, {}

    def _evaluate_cross(self, klines: dict, condition: dict, rule: dict) -> tuple[bool, dict]:
        """评估交叉类条件"""

        # 计算两个指标
        result1 = self.calculator.calculate(
            condition['indicator_key'], klines, rule.get('indicator_params', {})
        )
        result2 = self.calculator.calculate(
            condition['indicator_key2'], klines, rule.get('indicator2_params', {})
        )

        # 获取比较字段
        field1 = rule.get('field1', 'values')
        field2 = rule.get('field2', 'values')

        values1 = result1.get(field1, [])
        values2 = result2.get(field2, [])

        # 确保是列表
        if not isinstance(values1, list):
            values1 = result1.get('values', [])
        if not isinstance(values2, list):
            values2 = result2.get('values', [])

        # 获取最新两个值
        if len(values1) < 2 or len(values2) < 2:
            return False, {}

        prev_val1, curr_val1 = values1[-2], values1[-1]
        prev_val2, curr_val2 = values2[-2], values2[-1]

        # 过滤None值
        if None in [prev_val1, curr_val1, prev_val2, curr_val2]:
            return False, {}

        # 判断交叉
        triggered = False
        if rule['direction'] == 'down':  # 死叉：前一日val1>val2，当日val1<val2
            triggered = prev_val1 > prev_val2 and curr_val1 < curr_val2
        else:  # 金叉
            triggered = prev_val1 < prev_val2 and curr_val1 > curr_val2

        trigger_value = {
            'prev_val1': prev_val1,
            'curr_val1': curr_val1,
            'prev_val2': prev_val2,
            'curr_val2': curr_val2
        }

        return triggered, trigger_value

    def _evaluate_break(self, klines: dict, condition: dict, rule: dict) -> tuple[bool, dict]:
        """评估突破/跌破类条件"""

        # 计算指标
        result = self.calculator.calculate(
            condition['indicator_key'], klines, rule.get('indicator_params', {})
        )

        # 获取指标值
        field = rule.get('indicator_field', 'latest')

        if field == 'latest':
            indicator_value = result.get('latest', result.get('value'))
        else:
            values = result.get(field, [])
            indicator_value = values[-1] if values else None

        # 获取价格
        close = klines['close'][-1]

        # 判断突破/跌破
        triggered = False
        if rule['direction'] == 'down':  # 跌破
            triggered = close < indicator_value
        else:  # 突破
            triggered = close > indicator_value

        trigger_value = {
            'close': close,
            'indicator_value': indicator_value
        }

        return triggered, trigger_value

    def _evaluate_threshold(self, klines: dict, condition: dict, rule: dict) -> tuple[bool, dict]:
        """评估阈值类条件"""

        # 计算指标
        result = self.calculator.calculate(
            condition['indicator_key'], klines, rule.get('indicator_params', {})
        )

        # 获取指标值
        field = rule.get('indicator_field', 'latest')

        if field == 'latest':
            value = result.get('latest', result.get('value'))
        else:
            values = result.get(field, [])
            value = values[-1] if values else None

        threshold = rule['threshold']

        # 比较操作
        ops = {
            'gt': lambda v, t: v > t,
            'lt': lambda v, t: v < t,
            'ge': lambda v, t: v >= t,
            'le': lambda v, t: v <= t,
            'eq': lambda v, t: v == t,
        }

        triggered = ops[rule['compare_op']](value, threshold)

        trigger_value = {
            'indicator_value': value,
            'threshold': threshold
        }

        return triggered, trigger_value
```
